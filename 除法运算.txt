public class Test1 {
public static void main(String[] args) {
        System.out.println(5 / 2);//除法运算
        System.out.println(5.0 / 2);
        System.out.println(5 / 2.0);
        System.out.println((float) 5 / 2);
        System.out.println(5 / (float) 2);
        System.out.println((float) (5 / 2));
    }
}
